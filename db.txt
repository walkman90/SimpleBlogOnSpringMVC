CREATE TABLE Users 
(
id BIGINT GENERATED DEFAULT AS IDENTITY NOT NULL,
login VARCHAR(20),
email VARCHAR(50)
password VARCHAR(30),
image_url VARCHAR(255)
PRIMARY KEY (id)
 );
 
CREATE TABLE Tweets 
(
id BIGINT GENERATED DEFAULT AS IDENTITY NOT NULL,
user_id BIGINT,
text LONG VARCHAR, 
PRIMARY KEY (id)
 );
